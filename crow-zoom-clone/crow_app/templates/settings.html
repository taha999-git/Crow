themeOptions.forEach(option => { option.addEventListener('click', function() { const value = this.getAttribute('data-value'); themeInput.value = value; // Update UI themeOptions.forEach(opt => opt.classList.remove('selected')); this.classList.add('selected'); // Preview theme change if (window.themeManager) { window.themeManager.setTheme(value); } }); }); // Color theme selection const colorThemeOptions = document.querySelectorAll('.color-theme-option'); const colorThemeInput = document.getElementById('color-theme'); colorThemeOptions.forEach(option => { option.addEventListener('click', function() { const value = this.getAttribute('data-value'); colorThemeInput.value = value; // Update UI colorThemeOptions.forEach(opt => opt.classList.remove('selected')); this.classList.add('selected'); // Preview color theme change if (window.themeManager) { window.themeManager.setColorTheme(value); } }); }); // Color pickers const primaryColorPicker = document.getElementById('primary-color'); const primaryColorHex = document.getElementById('primary-color-hex'); const secondaryColorPicker = document.getElementById('secondary-color'); const secondaryColorHex = document.getElementById('secondary-color-hex'); if (primaryColorPicker && primaryColorHex) { primaryColorPicker.addEventListener('input', function() { primaryColorHex.value = this.value; }); primaryColorHex.addEventListener('input', function() { if (this.value.match(/^#[0-9A-F]{6}$/i)) { primaryColorPicker.value = this.value; } }); } if (secondaryColorPicker && secondaryColorHex) { secondaryColorPicker.addEventListener('input', function() { secondaryColorHex.value = this.value; }); secondaryColorHex.addEventListener('input', function() { if (this.value.match(/^#[0-9A-F]{6}$/i)) { secondaryColorPicker.value = this.value; } }); } // Toggle switches for preview const compactModeToggle = document.querySelector('input[name="compact_mode"]'); const animationsToggle = document.querySelector('input[name="animations_enabled"]'); const fontSizeSelect = document.querySelector('select[name="font_size"]'); if (compactModeToggle && window.themeManager) { compactModeToggle.addEventListener('change', function() { window.themeManager.setCompactMode(this.checked); }); } if (animationsToggle && window.themeManager) { animationsToggle.addEventListener('change', function() { window.themeManager.setAnimationsEnabled(this.checked); }); } if (fontSizeSelect && window.themeManager) { fontSizeSelect.addEventListener('change', function() { window.themeManager.setFontSize(this.value); }); } // Form submission for appearance settings const appearanceForm = document.querySelector('#appearance-tab'); if (appearanceForm) { const form = appearanceForm.closest('form'); if (form) { form.addEventListener('submit', function(e) { // Save current preview settings to localStorage temporarily // This ensures the theme persists until the page reloads if (window.themeManager) { const theme = themeInput.value; const colorTheme = colorThemeInput.value; localStorage.setItem('crow_theme', theme); localStorage.setItem('crow_color_theme', colorTheme); if (compactModeToggle) { localStorage.setItem('crow_compact_mode', compactModeToggle.checked); } if (animationsToggle) { localStorage.setItem('crow_animations', animationsToggle.checked); } if (fontSizeSelect) { localStorage.setItem('crow_font_size', fontSizeSelect.value); } } }); } } // Reset appearance button const resetBtn = document.querySelector('button[name="reset_appearance"]'); if (resetBtn) { resetBtn.addEventListener('click', function(e) { if (confirm('Are you sure you want to reset all appearance settings to defaults?')) { // Reset preview to defaults if (window.themeManager) { window.themeManager.setTheme('light'); window.themeManager.setColorTheme('blue'); window.themeManager.setCompactMode(false); window.themeManager.setAnimationsEnabled(true); window.themeManager.setFontSize('medium'); } // Reset form values themeInput.value = 'light'; colorThemeInput.value = 'blue'; primaryColorPicker.value = '#2563eb'; primaryColorHex.value = '#2563eb'; secondaryColorPicker.value = '#7c3aed'; secondaryColorHex.value = '#7c3aed'; // Update UI themeOptions.forEach(opt => { opt.classList.remove('selected'); if (opt.getAttribute('data-value') === 'light') { opt.classList.add('selected'); } }); colorThemeOptions.forEach(opt => { opt.classList.remove('selected'); if (opt.getAttribute('data-value') === 'blue') { opt.classList.add('selected'); } }); // Show success message alert('Appearance settings have been reset to defaults. Click "Save Changes" to apply.'); } }); } } // Initialize when tab is shown document.addEventListener('DOMContentLoaded', function() { const appearanceTab = document.getElementById('appearance-tab'); if (appearanceTab) { // Initialize when tab becomes active const observer = new MutationObserver(function(mutations) { mutations.forEach(function(mutation) { if (mutation.type === 'attributes' && mutation.attributeName === 'class') { if (appearanceTab.classList.contains('active')) { initAppearanceSettings(); } } }); }); observer.observe(appearanceTab, { attributes: true }); } });
